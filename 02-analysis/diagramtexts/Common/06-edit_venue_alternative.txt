control VenueService

entity TicketType

Actor ->> VenueSystem : updateVenue(venueDetails: VenueDetails)

note right of VenueSystem:2. The system displays a form with\ the current venue details (27-edit_venue.png).
    VenueSystem ->> VenueService : updateVenue(venueDetails: VenueDetails)
    activate VenueService

 note right of VenueService:3. The organiser or administrator updates the information\n(Name, Address, Ticket Types(ticketTypeName, number),\nContact information (name, phone, email)).

        Venue->>TicketType:TicketType(ticketTypeName,number)

    VenueService ->> VenueService : checkDuplicateVenue(v)
    activate VenueService
    VenueService --> VenueService : return isDuplicate
    deactivate VenueService

    alt isDuplicate == true
        VenueService -->> VenueSystem : return false
deactivate VenueSystem
deactivate VenueService
end
