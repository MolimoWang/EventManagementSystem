control ": AccountSystem" as acs
participant ": AccountService" as asvc

== 1. The administrator selects 'View All Accounts' ==
admin -> acs: viewAccounts(accountEmail)
activate acs

== 2. The system retrieves all existing accounts ==
acs -> asvc: getAccounts()
activate asvc

loop for every account
  asvc -> acc: readAccount(accountDetails)
end

asvc --> acs: return all accounts
deactivate asvc

acs -> acs: updateDisplay()

acs --> admin:
deactivate acs
