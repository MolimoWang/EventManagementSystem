@startuml
actor User
participant "AccountSystem" as AS
participant "AccountService" as ACS
participant "Account" as A



== Purchase Ticket (Insufficient Tickets) ==
User -> AS: select 'Buy it!' option for an event
AS -> User: ask for confirmation
User -> AS: confirms Yes
AS -> ACS: checkTicketAvailability(userEmail, eventName)
alt No Tickets Available
AS -> User: display message "Tickets are sold out"
end

@enduml
